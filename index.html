import java.util.Scanner;

public class DS2Tarea3 {
   public static void main(String args[]){
            int z;
            sumar();
            desplegar(5);
            z = cuadrado(10);
            System.out.println(z);
            raizCuadrada();
   }
   
   public static void sumar() {
        int n1,n2,suma;
        Scanner sc = new Scanner(System.in);   
        System.out.print("Valor 1 : ");
        n1 = sc.nextInt();
        System.out.print("Valor 2 : ");
        n2 = sc.nextInt();
        suma = n1 + n2;
        System.out.println("La suma es : " + suma);
   }
   
   public static void desplegar (int x) {
       int i;
       for (i=1;i<=x;i++) {
           System.out.print(i+" ");
       }
       System.out.println("");
   }
   public static int cuadrado (int x) {
       int resultado;
       resultado = x * x;
       return resultado;
   }
   public static void raizCuadrada () {
       double resultado,n;
       Scanner sc = new Scanner(System.in);
       System.out.print("Ingresa un valor: ");
       n = sc.nextDouble();
       resultado = Math.pow(n, 0.5);
       System.out.println("La raíz cuadrada es: " + resultado);
   }

  }
